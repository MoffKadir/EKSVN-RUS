#свекровь/тёща в зависимости от пола игрока (вызывается через [Player.])
defined_text = {
	name = GetMotherInLaw

	text = {
		trigger = {
			NOT = { character = FROM }
			is_female = yes
		}
		localisation_key = GetRelation_MotherOfHusband
	}
	text = {
		trigger = {
			NOT = { character = FROM }
			is_female = no
		}
		localisation_key = GetRelation_MotherOfWife
	}
	fallback_text = {
		localisation_key = GetMotherInLaw_FB
	}
}
